#CSS #라이브러리
# 개요






# css()
``` js
<div
  className={css({
    display: "flex",
    flexDirection: "column",
    bgColor: "surfaceSecondary",
    h: "100vh",
  })}
>
```



# stack()
``` js
<div
  className={stack({
    gap: 0,
    w: "100%",
    h: "100%",
    alignItems: "center",
    justifyContent: "center",
  })}
>
```



# flex()
``` js
<div
  className={flex({
    justifyContent: "flex-end",
    gap: "9px",
  })}
>
```
